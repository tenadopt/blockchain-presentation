Example Solidity Snippet


pragma solidity ^0.8.0;

contract Payment {
    address payable public seller;

    constructor() {
        seller = payable(msg.sender);
    }

    function pay() public payable {
        require(msg.value > 0, "No funds sent");
        seller.transfer(msg.value);
    }
}
